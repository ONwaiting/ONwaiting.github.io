更新个人信息 #

示例

js
const response = await mxSdk.request.http({ 
    url: '/friends/user/v1/updatePlayerInfo',
    /** 请求入参 */
    data:requestData
 })

requestData 请求入参格式

ts
type requestData = {
    /** 用户id */
    gameId: string,
    /** 要更新的用户信息 增量更新。如果使用推荐功能，玩家应至少更新一次推荐分数字段后才能使用推荐功能 */
    info?: FriendMessagePlayerDetailsInfo
}
type FriendMessagePlayerDetailsInfo = {
    /** 扩展字段 */
    extra?: { [key: string]: string },
    /** 昵称 */
    nickname?: string,
    /** 头像 */
    avatar?: string,
    /**
     * 额外的好友数量上限
     */
    extraFriendNum?: number,
}

response 返回参数格式

ts
type response = {
    /** 状态码 */
    code: number,
    /** 服务器时间 */
    serverTime?: number,
    /** 更新后的用户详细信息 */
    info?: FriendMessagePlayerDetailsInfo
}

type FriendMessagePlayerDetailsInfo = {
    /** 扩展字段 */
    extra?: { [key: string]: string },
    /** 昵称 */
    nickname?: string,
    /** 头像 */
    avatar?: string,
    /**
     * 额外的好友数量上限
     */
    extraFriendNum?: number,
}
